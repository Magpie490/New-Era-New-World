take_triumphant = {
	name = CB_NAME_TAKE_PIRATE_KING
	war_name = WAR_NAME_TAKE_PIRATE_KING
	sprite = 21
	truce_days = 1825
	is_permanent = yes
	can_ask_to_join_war = yes
	check_all_titles = yes
	
	can_use = {
		ROOT = {
			religion = triumphant
			has_global_flag = warlike_triumphant_flag
		}
	}
	
	is_valid = {
		ROOT = {
			religion = triumphant
		}
	}
	can_use_title = {
		title = d_triumphant
	}
	on_add = {
		general_on_add_effect = yes
	}
	on_success = {
		any_attacker = {
			show_scope_change = no
			limit = { character = ROOT }
			participation_scaled_piety = 150
			participation_scaled_prestige = 250
		}
		any_attacker = {
			limit = { NOT = { character = ROOT } }
			hidden_tooltip = { 
				participation_scaled_piety = 150
				participation_scaled_prestige = 250
			}
		}
		FROM = {
			show_scope_change = no
			piety = -150
			prestige = -250
		}
	}
	on_success_title = {
		usurp_title = ROOT
	}
	on_fail = {
		ROOT = {
			piety = -75
			prestige = -125
		}
		FROM = {
			show_scope_change = no
			piety = 75
			prestige = 125
		}
	}

	on_reverse_demand = {
		any_defender = {
			show_scope_change = no
			limit = { character = FROM }
			participation_scaled_piety = 200
			participation_scaled_prestige = 300
		}
		any_defender = {
			limit = { NOT = { character = FROM } }
			hidden_tooltip = {
				participation_scaled_piety = 200
				participation_scaled_prestige = 300
			}
		}
		ROOT = {
			show_scope_change = no
			transfer_scaled_wealth = {
				to = FROM
				value = 4.0
			}
			piety = -200
			prestige = -300
			imprison = FROM
		}
	}

	attacker_ai_victory_worth = {
		factor = -1 # always accept
	}
	
	attacker_ai_defeat_worth = {
		factor = 100
	}

	defender_ai_victory_worth = {
		factor = -1 # always accept
	}
	
	defender_ai_defeat_worth = {
		factor = 100
	}
}

exovedate_invasion = {
	name = CB_NAME_INVASION
	war_name = WAR_NAME_INVASION
	sprite = 8
	truce_days = 3650
	hostile_against_others = yes
	is_permanent = yes
	can_ask_to_join_war = yes
	check_de_jure_tier = KING # this scans all de jure kingdoms for the counties which are held by or vassals(or below) of selected character. Only valid if is_permanent = yes
	
	allowed_to_target_tributaries = no
	
	sort_priority = 2000

	can_use_gui = {
		ROOT = {
			prestige = 500
		}
		check_if_crusader_trigger = yes
	}
	
	can_use = {
	}
	
	on_add = {
		ROOT = { show_scope_change = no prestige = -500 }
		general_on_add_effect = yes
		if = {
			limit = { 
				attacker = { 
					any_owned_bloodline = {
						has_bloodline_flag = bloodline_great_conquerors
					} 
				} 
			}
			attacker = {  
				set_character_flag = flag_declared_bloodline_invasion
			}
		}
	}

	can_use_title = {
		ROOT = {
			OR = {
				ai = yes
				multiplayer = no
				NOT = {
					has_game_rule = {
						name = multiplayer_invasions
						value = off
					}
				}
			}
			is_nomadic = no
			OR = {
				AND = {
					has_character_flag = horde_invader
					independent = yes
					higher_tier_than = COUNT
				}
				AND = {
					mercenary = yes
					NOT = {
						num_of_titles = 2
					}
				}
				AND = {
					has_character_flag = is_child_of_destiny
					any_owned_bloodline = {
						has_bloodline_flag = founder_invasions
					}
				}
				AND = {
					NOT = { has_character_flag = flag_declared_bloodline_invasion }
					any_owned_bloodline = {
						has_bloodline_flag = bloodline_great_conquerors
					}
				}
				AND = {
					culture_group = droid
					religion_group = synth_group
				}
				AND = {
					primary_title = {
					title = e_druze
					}
				}
			}
		}
		
		ROOT = {
			NOT = { same_realm = FROM }
		}
		
		OR = { # This ensures that the AI doesn't declare this type of war against someone who only holds a few counties within the targeted title...
			ROOT = {
				ai = no
			}
			ROOT = {
				has_character_flag = horde_invader
			}
			FROM = {
				num_of_realm_counties = {
					value = 3
					title = PREV
				}
			}
		}
	}
	
	is_valid_title = {
		FROM = {
			any_realm_province = {
				de_jure_liege_or_above = PREVPREV
			}
		}
	}

	on_success = {
		ROOT = {
			show_scope_change = no
			prestige = 200
			participation_scaled_prestige = 200
			narrative_event = { id = exovedate.3 }
			grant_title = k_grand_montreal
		}
		any_attacker = {
			limit = { NOT = { character = ROOT } }
			hidden_tooltip = { participation_scaled_prestige = 200 }
		}
		FROM = {
			show_scope_change = no
			prestige = -200
		}
		hidden_tooltip = {
			ROOT = {
				occupy_minors_of_occupied_settlements = FROM
				gain_all_occupied_titles = { who = FROM type = invasion }
				
				if = {
					limit = {
						has_nickname = no
					}
					random_list = {
						10 = { give_nickname = nick_the_great }
						10 = { give_nickname = nick_the_victorious }
						10 = { give_nickname = nick_the_conqueror }
						10 = { give_nickname = nick_the_brave }
						10 = { give_nickname = nick_the_dragon }
						10 = { give_nickname = nick_the_terrible }
					}
				}
			}
		}
	}
	
	on_success_title = {
		
		custom_tooltip = {
			text = tribal_invasion_succ_tip
			hidden_tooltip = {
				ROOT = {
					vassalize_or_take_under_title_destroy_duchies = {
						title = PREV
						enemy = FROM
